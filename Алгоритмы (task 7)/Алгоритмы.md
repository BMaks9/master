```
def sum_rec(n: int):
	if n == 1:
		return 1
	return 1 / (n ** 2) + sum_rec(n - 1)
```

```
def foo(text: str, ch: str) -> str:
	text_lw = text.lower()
	ch_lw = ch.lower()
	
	first = text_lw.find(ch_lw)
	end = text_lw.rfind(ch_lw)
	
	if first == -1 or first == end:
		return text
	
	return text[:first] + text[first:end+1][::-1] + text[end+1:]
```

6. Матрица + суммы относительно диагонали
![[Pasted image 20260717030203.png]]
Очень простая задача, требуется только понимание того как определить положение элементов относительно диагонали. 

Задача по сути решается в ф-ии foo(), все остальное дроч с инпутами, оформлением и проверками, которые можно писать бесконечно и которые занимают 80% времени от написания решения.

Ответ:
```
import random

def foo(matrix: list[list[int]], k: int, m: int):
	sum_low = sum_down = 0
	diagonal = k - m
	
	for i, row in enumerate(matrix):
		for j, value in enumerate(row):
			if i - j > diagonal:
				sum_down += value
			elif i - j < diagonal:
				sum_low += value
	
	return sum_low, sum_down

if __name__ == '__main__':
	n = int(input("Input N - "))
	k = int(input("Input k - "))
	m = int(input("Input m - "))
	
	while m >= n:
		print("Incorrect m - ")
		m = int(input("Input m - "))
		
	matrix = [[random.randint(-10**5, 10**5)for _ in range(n)] for _ in range(n)]
	low, down = foo(matrix, k, m)
	
	for row in matrix:
		print(*row, sep='\t')
	print(low, down)

```

7. Матрица + среднее ариф. СТОЛБЦА
![[Pasted image 20260717231142.png]]

Ответ:
```
import random

def foo(matrix: list[list[int]]) -> list[list[int]]:
	result = [row.copy() for row in matrix]
	rows = len(matrix)
	columns = len(matrix[0])
	
	for j in range(columns):
		column = [matrix[i][j] for i in range(rows)]
		result[0][j] = (max(column) + min(column)) / 2
	return result

if __name__ == '__main__':

		print(*row, sep='\t')

		print(*row, sep='\t')
```

8.  Матрица, уменьшаем 
![[Pasted image 20260717234058.png]]

Очень просто и очень скучно

Ответ:
```
import random

def foo(matrix: list[list[int]]) -> list[list[int]]:
	result = [row.copy() for row in matrix]
	count = 0
	total = 0
	
	for row in result:
		for value in row:
			if value > 0:
				total += value
				count+=1
	
	if count == 0:
		return result
		
	avg = total / count
	
	for i, row in enumerate(result):
		for j, _ in enumerate(row):
			result[i][j] -= avg
	return result 

if __name__ == '__main__':
	n = int(input("Input N - "))
```

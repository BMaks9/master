def sum_rec(n: int):
	if n == 1:
		return 1
	return 1 / (n ** 2) + sum_rec(n - 1)
```

2. Строка
![[Pasted image 20260716001700.png]]

Ответ:
```
def foo(text: str, ch: str) -> str:
	text_lw = text.lower()
	ch_lw = ch.lower()
	
	first = text_lw.find(ch_lw)
	end = text_lw.rfind(ch_lw)
	
	if first == -1 or first == end:
		return text
	
	return text[:first] + text[first:end+1][::-1] + text[end+1:]
```

3. Локальные максимумы
![[Pasted image 20260716005506.png]]

Ответ:
```
import random

def foo(n: int):
	data = [random.randint(1, 10**5) for _ in range(n)]
	count = 0
	
	for i in range(1, len(data) - 1):
		if data[i-1] < data[i] > data[i+1]:
			count += 1
	
	return data, count	
```

import random

def foo(n: int):
	data = [random.randint(1, 10**5) for _ in range(n)]
	direction = 0
	count = 1
	max_count = 1
	
	for i in range(1, len(data)):
		if data [i - 1] < data[i]:
			new_direction = 1
		elif data [i - 1] > data[i]:
			new_direction = -1
		else:
			direction = 0
			count = 1
			continue
		
		if direction == new_direction:
			count += 1
		else:
			count = 2 # как минимум будет одна пара из 2 эл.
			direction = new_direction
		
		max_count = max(max_count, count)
	return data, max_count
```
